3   /**
4    * This class holds the description, tag (used in SIF file), the information if the interaction is
5    * directed for the relevant binary interaction types.
6    */
7   public enum BinaryInteractionType
8   {
9   
10  	COMPONENT_OF(
11  			"The first entity is a component of the second entity, which is a complex.  " +
12  			"This interaction is transitive in the sense that A component_of B and B component_of " +
13  			"C implies A component_of C.  " +
14  			"This interaction is directed.", true),
15  
16  	IN_SAME_COMPONENT(
17  			"Two entities belong to the same molecular complex.  This does not necessarily " +
18  			"mean they interact directly.  In a complex with n molecules, this rule will create a " +
19  			"clique composed of n(n-1)/2 interactions.  " +
20  			"This interaction is undirected.", false),
21  
22  	CO_CONTROL(
23  			"This rule infers an interaction " +
24  			"if the first and second entities have control over the same process. " +
25  			"This interaction is undirected.", false),
26  
27  	SEQUENTIAL_CATALYSIS(
28  			"The entities catalyze two conversions that are connected via a common molecule, " +
29  			"e.g. the first entity produces a substrate that is consumed by the second entity.  " +
30  			"This interaction is directed.", true),
31  
32  	METABOLIC_CATALYSIS(
33  			"The first entity catalyzes a reaction that either consumes or produces the second entity.  " +
34  			"This interaction is directed.", true),
35  
36  	STATE_CHANGE(
37  			"The first entity controls a reaction that changes the state of the " +
38  			"second entity, e.g. by phosphorylation or other posttranslational modification, " +
39  			"or by a change in subcellular location.  " +
40  			"This interaction is directed.", true),
41  
42  	ACTIVATES(
43  			"The first entity controls a reaction that changes the state of the " +
44  			"second entity, from inactive or notr, to active. " +
45  			"This interaction is directed.", true),
46  
47  	INACTIVATES(
48  			"The first entity controls a reaction that changes the state of the " +
49  			"second entity, from active or notr, to inactive. " +
50  			"This interaction is directed.", true),
51  
52  	REACTS_WITH(
53  			"The entities participate in a conversion as substrates or products.  Controllers are not included.  " +
54  			"This interaction is undirected.", false),
55  
56  	INTERACTS_WITH(
57  			"The entities participate in an interaction.  Controllers are not included.  " +
58  			"This interaction is undirected.", false),
59  
60  	UPREGULATE_EXPRESSION(
61  			"The first entity upregulates the expression of the second entity. " +
62  			"This interaction is directed.", true),
63  
64  	DOWNREGULATE_EXPRESSION(
65  			"The first entity downregulates the expression of the second entity. " +
66  			"This interaction is directed.", true),
67  
68  	GENERIC_OF(
69  			"The first entity is a generic form of the second entity.This interaction is directed",true),
70  
71  	SIMILAR(
72  			"The two entities are members of a generic form. This interaction is undirected",false);
73  
74  	private String description;
75  	private boolean directed;
76  
77  	BinaryInteractionType(
78  			String description,
79  			boolean directed)
80  	{
81  		this.description = description;
82  		this.directed = directed;
83  	}
84  
85  	/**
86  	 * Returns the description of the binary interaction as a string.
87  	 *
88  	 * @return a string describing the interaction
89  	 */
90  	public String getDescription()
91  	{
92  		return description;
93  	}
94  
95  	/**
96  	 * Returns the tag of the binary interaction. This tag is used in the SIF file(s).
97  	 *
98  	 * @return tag name as it is used in SIF files
99  	 */
100 	public String getTag()
101 	{
102 		return this.toString();
103 	}
104 
105 	/**
106 	 * Returns true, if the interaction is directed.
107 	 *
108 	 * @return false, if the interaction is not directed.
109 	 */
110 	public boolean isDirected()
111 	{
112 		return directed;
113 	}
114 }
