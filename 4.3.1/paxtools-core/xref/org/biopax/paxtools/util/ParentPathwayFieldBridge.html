4   package org.biopax.paxtools.util;

6   import java.util.Set;

8   import org.apache.lucene.document.Document;
9   import org.biopax.paxtools.model.level3.Pathway;
10  import org.biopax.paxtools.model.level3.UnificationXref;
11  import org.biopax.paxtools.model.level3.Xref;
12  import org.hibernate.search.bridge.FieldBridge;
13  import org.hibernate.search.bridge.LuceneOptions;
14  
15  /**
16   * This is a {@link FieldBridge} implementation 
17   * for biopax elements to generate a custom search/filter 
18   * index field that contain their parent pathway URIs and names. 
19   * 
20   * @author rodche
21   * @deprecated Hibernate ORM/Search will be removed in v5
22   */
23  public final class ParentPathwayFieldBridge implements FieldBridge {
24  
25  	@Override
26  	public void set(String name, Object value, Document document, LuceneOptions luceneOptions) {
27  		if(value instanceof Set) {
28  			for(Pathway pw : (Set<Pathway>) value) {
29  				//adding URI as is (do not change the URI to lowercase!)
30  				FieldBridgeUtils.addFieldToDocumentAsIs(luceneOptions, name, pw.getRDFId(), document);
31  				// add names to the index as well
32  				for (String s : pw.getName()) {
33  					FieldBridgeUtils.addFieldToDocument(luceneOptions, name, s, document);
34  				}
35  				// add unification xref IDs too
36  				for (UnificationXref x : new ClassFilterSet<Xref, UnificationXref>(
37  						pw.getXref(), UnificationXref.class)) {
38  					if (x.getId() != null)
39  						FieldBridgeUtils.addFieldToDocument(luceneOptions, name, x.getId(), document);
40  				}
41  			}
42  		} else {
43  			throw new AssertionError("Bug!");
44  		}
45  	}
46  }
