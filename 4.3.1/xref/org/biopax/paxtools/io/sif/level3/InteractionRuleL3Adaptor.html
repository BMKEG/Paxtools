14  /**
15   * Base class for level 3 interactions rules.
16   */
17  public abstract class InteractionRuleL3Adaptor implements InteractionRuleL3
18  {
19  	/**
20  	 * Log for logging.
21  	 */
22  	private final Log log = LogFactory.getLog(ParticipatesRule.class);
23  
24  	/**
25  	 * Selects only PhysicalEntity as seed.
26  	 * @param interactionSet inferred interactions
27  	 * @param entity this must be a PhysicalEntity for L3
28  	 * @param model BioPAX model
29  	 */
30  	public final void inferInteractions(InteractionSet interactionSet, BioPAXElement entity,
31  		Model model)
32  	{
33  		if (entity instanceof PhysicalEntity)
34  		{
35  			inferInteractionsFromPE((InteractionSetL3) interactionSet, ((PhysicalEntity) entity),
36  				model);
37  		}
38  		else
39  		{
40  			if (log.isInfoEnabled()) log.info("Not a PE Skipping." + entity.getRDFId());
41  		}
42  	}
43  
44  	/**
45  	 * Initializes the options.
46  	 * @param options options map
47  	 */
48  	public void initOptions(Map options)
49  	{
50  		if (options == null)
51  		{
52  			options = new HashMap();
53  		}
54  		initOptionsNotNull(options);
55  	}
56  
57  	/**
58  	 * Does nothing for this class. Initializes the options where this is overwritten.
59  	 * @param options options map
60  	 */
61  	protected void initOptionsNotNull(Map options)
62  	{
63  	}
64  
65  	/**
66  	 * Checks if the options map has the given option with the given value.
67  	 * @param key key of the option
68  	 * @param value value of the option
69  	 * @param options options map
70  	 * @return true if the options map has the given option with the given value
71  	 */
72      protected boolean checkOption(Object key, Object value, Map options)
73      {
74          return options.containsKey(key)&&options.get(key).equals(value);
75      }
76  
77  	/**
78  	 *
79  	 * @param pes physical entities
80  	 * @param set inferred binary interactions
81  	 * @return BioPAX objects
82  	 */
83  	protected Set<BioPAXElement> collectEntities(Set<PhysicalEntity> pes, InteractionSetL3 set)
84  	{
85  		Set<BioPAXElement> entities = new HashSet<BioPAXElement>();
86  		for (PhysicalEntity pe : pes)
87  		{
88  			BioPAXElement entity = set.getGroupMap().getEntityReferenceOrGroup(pe);
89  			if (entity != null) entities.add(entity);
90  			if (entity instanceof Group)
91  			{
92  				getMembersRecursively(entities, (Group) entity);
93  			}
94  		}
95  		return entities;
96  	}
97  
98  	/**
99  	 * Gets members of the Group recursively traversing sub-Groups.
100 	 * @param entities member set
101 	 * @param group group to query
102 	 */
103 	private void getMembersRecursively(Set<BioPAXElement> entities, Group group)
104 	{
105 		entities.addAll(group.members);
106 		for (Group subgroup : group.subgroups)
107 		{
108 			getMembersRecursively(entities, subgroup);
109 		}
110 	}
111 
112 	/**
113 	 * Creates an interactions between every ordered pair of components.
114 	 * @param interactionSet inferred interactions
115 	 * @param components nodes of the clique
116 	 * @param type interaction type
117 	 * @param mediators mediator elements of te interaction
118 	 */
119 	protected void createClique(InteractionSetL3 interactionSet, List<BioPAXElement> components,
120 			BinaryInteractionType type, BioPAXElement... mediators)
121 	{
122         GroupMap groupMap = interactionSet.getGroupMap();
123 
124 		for (int j = 0; j < components.size(); j++)
125 		{
126 			for (int i = 0; i < j; i++) {
127                 createAndAdd(
128                         groupMap.getEntityReferenceOrGroup(components.get(i)),
129                         groupMap.getEntityReferenceOrGroup(components.get(j)), interactionSet,
130                         type,
131                         mediators);
132             }
133         }
134 	}
135 
136 	/**
137 	 * Creates the binary interaction and adds to the inferred interaction list.
138 	 * @param source source of the interaction
139 	 * @param target target of the interaction
140 	 * @param is3 inferred interactions
141 	 * @param type interaction type
142 	 * @param mediators mediators of the interaction
143 	 */
144     protected void createAndAdd(
145             BioPAXElement source,
146             BioPAXElement target,
147             InteractionSetL3 is3,
148             BinaryInteractionType type,
149             BioPAXElement... mediators)
150     {
151         if(source!=null && target!=null && !source.equals(target))
152         {
153             SimpleInteraction sc = new SimpleInteraction(source, target, type);
154             for (BioPAXElement mediator : mediators) {
155                 sc.addMediator(mediator);
156             }
157             is3.add(sc);
158         }
159     }
