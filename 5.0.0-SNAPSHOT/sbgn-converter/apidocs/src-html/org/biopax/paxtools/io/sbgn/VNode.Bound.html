085     * @param g glyph
086     * */
087    public VNode(Glyph g)
088    {
089        SOURCE_AND_SINK_BOUND = new Bound(15,15);
090        LOGICAL_OPERATOR_BOUND = new Bound(15,15);
091        PROCESS_NODES_BOUND = new Bound(15,15);
092
093        MACROMOLECULE_BOUND = new Bound(48,25);
094        NUCLEIC_ACID_FEATURE_BOUND = new Bound(50,25);
095
096        SIMPLE_CHEMICAL_BOUND = new Bound(48,20);
097        UNSPECIFIED_ENTITY_BOUND = new Bound(40,40);
098        PHENOTYPE_BOUND = new Bound(50,20);
099        TAG_BOUND = new Bound(50,20);
100        PERTURBING_AGENT_BOUND = new Bound(50,20);
101        COMPLEX_BOUND = new Bound(48,20);
102
103        INFO_BOUND = new Bound(MAX_STATE_AND_INFO_WIDTH,MAX_STATE_AND_INFO_HEIGHT);
104        STATE_BOUND = new Bound(MAX_STATE_AND_INFO_WIDTH,MAX_STATE_AND_INFO_HEIGHT);
105
106        stateGlyphs = new ArrayList<Glyph>();
107        infoGlyphs = new  ArrayList<Glyph>();
108
109        this.glyph = g;
110
111        /*
112                 * need to add bbox objects
113                 */
114        Bbox b = new Bbox();
115        this.glyph.setBbox(b);
116
117        if (this.glyph.getClazz() == null)
118            this.glyph.setClazz(NONE);
119
120        this.setSizeAccordingToClass();
121    }

123    /**
124     * Function that will take place when VNode objects will update in layout process of ChiLay
125     *
126     * @param lGraphObj LGraphObject for whom the update will take place.
127     */
128    public void update(LGraphObject lGraphObj)
129    {
130        if (lGraphObj instanceof CoSEGraph)
131        {
132            return;
133        }
134
135        LNode lNode = (LNode)lGraphObj;
136
137        this.glyph.getBbox().setX((float) lNode.getLeft());
138        this.glyph.getBbox().setY((float) lNode.getTop());
139        this.placeStateAndInfoGlyphs();
140    }

153    /**
154     * Chooses a proper bound for this VNode according to its class.
155     */
156    public void setSizeAccordingToClass()
157    {
158        String glyphClass = this.glyph.getClazz();
159
160        //If glyph class is not specified return here
161        if (glyphClass.equalsIgnoreCase(NONE))
162            return;
163
164        if (glyphClass.equalsIgnoreCase(SOURCE_AND_SINK))
165        {
166            setBounds(SOURCE_AND_SINK_BOUND.getWidth(), SOURCE_AND_SINK_BOUND.getHeight());
167        }
168
169        else if (glyphClass.equalsIgnoreCase(AND) || glyphClass.equalsIgnoreCase(OR) || glyphClass.equalsIgnoreCase(NOT))
170        {
171            setBounds(LOGICAL_OPERATOR_BOUND.getWidth(), LOGICAL_OPERATOR_BOUND.getHeight());
172        }
173
174        else if (glyphClass.equalsIgnoreCase(ASSOCIATION) || glyphClass.equalsIgnoreCase(DISSOCIATION) || glyphClass.equalsIgnoreCase(OMITTED_PROCESS) ||
175                glyphClass.equalsIgnoreCase(UNCERTAIN_PROCESS) || glyphClass.equalsIgnoreCase(PROCESS))
176        {
177            setBounds(PROCESS_NODES_BOUND.getWidth(), PROCESS_NODES_BOUND.getHeight());
178        }
179
180        else if (glyphClass.equalsIgnoreCase(SIMPLE_CHEMICAL))
181        {
182            setBounds(SIMPLE_CHEMICAL_BOUND.getWidth(), SIMPLE_CHEMICAL_BOUND.getHeight());
183        }
184
185        else if (glyphClass.equalsIgnoreCase(UNSPECIFIED_ENTITY))
186        {
187            setBounds(UNSPECIFIED_ENTITY_BOUND.getWidth(), UNSPECIFIED_ENTITY_BOUND.getHeight());
188        }
189
190        else if (glyphClass.equalsIgnoreCase(MACROMOLECULE))
191        {
192            setBounds(MACROMOLECULE_BOUND.getWidth(), MACROMOLECULE_BOUND.getHeight());
193        }
194
195        else if (glyphClass.equalsIgnoreCase(NUCLEIC_ACID_FEATURE))
196        {
197            setBounds(NUCLEIC_ACID_FEATURE_BOUND.getWidth(), NUCLEIC_ACID_FEATURE_BOUND.getHeight());
198        }
199
200        else if (glyphClass.equalsIgnoreCase(STATE_VARIABLE))
201        {
202            setBounds(STATE_BOUND.getWidth(), STATE_BOUND.getHeight());
203        }
204
205        else if (glyphClass.equalsIgnoreCase(UNIT_OF_INFORMATION))
206        {
207            setBounds(INFO_BOUND.getWidth(), INFO_BOUND.getHeight());
208        }
209
210        else if (glyphClass.equalsIgnoreCase(PHENOTYPE))
211        {
212            setBounds(PHENOTYPE_BOUND.getWidth(), PHENOTYPE_BOUND.getHeight());
213        }
214
215        else if (glyphClass.equalsIgnoreCase(PERTURBING_AGENT))
216        {
217            setBounds(PERTURBING_AGENT_BOUND.getWidth(), PERTURBING_AGENT_BOUND.getHeight());
218        }
219
220        else if (glyphClass.equalsIgnoreCase(TAG))
221        {
222            setBounds(TAG_BOUND.getWidth(), TAG_BOUND.getHeight());
223        }
224
225        else if (glyphClass.equalsIgnoreCase(COMPLEX))
226        {
227            setBounds(COMPLEX_BOUND.getWidth(), COMPLEX_BOUND.getHeight());
228        }
229
230        if( this.glyph.getClone() != null )
231        {
232            Bbox glyphBbox = this.glyph.getBbox();
233            setBounds(3*glyphBbox.getW()/4, 3*glyphBbox.getH()/4);
234        }
235
236        if (glyphClass.equalsIgnoreCase(MACROMOLECULE) || glyphClass.equalsIgnoreCase(NUCLEIC_ACID_FEATURE) || glyphClass.equalsIgnoreCase(SIMPLE_CHEMICAL) || glyphClass.equalsIgnoreCase(COMPLEX))
237        {
238            updateSizeForStateAndInfo();
239        }
240
241    }

243    /**
244     * Calculates required width according to the given list state and info glyphs of this VNode.
245     * This method also previously computes the width and height of state and info glyphs
246     * according to their label.
247     *
248     * @param  stateORinfoList list that keeps state or info glyphs of this VNode
249     * @return new width that is adjusted so that all glyphs in stateORinfoList are included.
250     * */
251    public int calcReqWidthByStateAndInfos(List<Glyph> stateORinfoList)
252    {
253        int wholeSize = 0;
254        int count = 0;
255
256        for (Glyph tmpGlyph: stateORinfoList)
257        {
258            String text;
259
260            if (tmpGlyph.getState() != null)
261            {
262                text = tmpGlyph.getState().getValue();
263
264                if (tmpGlyph.getState().getVariable() != null &&
265                        tmpGlyph.getState().getVariable().length() > 0)
266                {
267                    if(tmpGlyph.getState().getVariable() != null)
268                        text += "@" + tmpGlyph.getState().getVariable();
269                }
270            }
271            else if (tmpGlyph.getLabel() != null)
272            {
273                text = tmpGlyph.getLabel().getText();
274            }
275            else
276            {
277                throw new RuntimeException("Encountered an information glyph with no state " +
278                        "variable (as modification boxes should have) and no label (as molecule type " +
279                        "boxed should have). glyph = " + tmpGlyph);
280            }
281
282            int numOfUpper = 0;
283            int numOfLower = 0;
284
285            for (int i = 0; i < text.length(); i++)
286            {
287                if (Character.isLowerCase(text.charAt(i)))
288                {
289                    numOfLower++;
290
291                } else
292                    numOfUpper++;
293            }
294
295            Bbox b = new Bbox();
296            tmpGlyph.setBbox(b);
297
298            //Set width
299            float requiredSize = numOfLower * LOWERCASE_LETTER_PIXEL_WIDTH + numOfUpper * UPPERCASE_LETTER_PIXEL_WIDTH;
300            if (requiredSize < MAX_STATE_AND_INFO_WIDTH)
301                tmpGlyph.getBbox().setW(requiredSize);
302            else
303                tmpGlyph.getBbox().setW(STATE_BOUND.width);
304
305            //Set height
306            tmpGlyph.getBbox().setH(MAX_STATE_AND_INFO_HEIGHT);
307
308            if (count < MAX_INFO_BOX_NUMBER / 2)
309                wholeSize += tmpGlyph.getBbox().getW();
310
311            count++;
312
313        }
314
315        return wholeSize;
316    }

318    /**
319     *  If glyph attribute of this VNode object includes any "state of information" or "unit of information" glyphs, this method updates the
320     *  size of VNode accordingly.
321     * */
322    public void updateSizeForStateAndInfo()
323    {
324        // Find all state and info glyphs
325        for (Glyph glyph : this.glyph.getGlyph())
326        {
327            if (glyph.getClazz() == STATE_VARIABLE)
328            {
329                stateGlyphs.add(glyph);
330            }
331            else if (glyph.getClazz() == UNIT_OF_INFORMATION)
332            {
333                infoGlyphs.add(glyph);
334            }
335        }
336
337        //Calculate "state of information" glyphs' sizes
338        int wholeWidthOfStates = calcReqWidthByStateAndInfos(stateGlyphs);
339        int wholeWidthOfInfos  = calcReqWidthByStateAndInfos(infoGlyphs);
340
341        // Calculate  positions
342        int numOfStates = stateGlyphs.size();
343        int numOfInfos = infoGlyphs.size();
344
345
346
347        //set a maximum width to glyph according to state and info boxes
348        int totNumStateInfo = numOfInfos + numOfStates;
349        int multiplier = totNumStateInfo <= 2 ? 2 : 3;
350        float requiredWidth = multiplier * OFFSET_BTW_INFO_GLYPHS + (multiplier-1) * MAX_STATE_AND_INFO_WIDTH;
351
352        //Adjust heights so that info box offsets are taken into account in layout.
353        if(totNumStateInfo > 0 )
354            this.glyph.getBbox().setH(this.glyph.getBbox().getH()+MAX_STATE_AND_INFO_HEIGHT/2);
355
356        if (this.glyph.getBbox().getW() < requiredWidth  )
357        {
358            this.glyph.getBbox().setW(requiredWidth);
359        }
360    }

362    /**
363     * Places state and info glyphs of this node
364     * */
365    public void placeStateAndInfoGlyphs() {
366        int numOfStates = stateGlyphs.size();
367        int numOfInfos = infoGlyphs.size();
368
369
370        //Adjust heights so that info box offsets are taken into account in layout.
371        if(numOfStates > 0 || numOfInfos > 0)
372            this.glyph.getBbox().setH(this.glyph.getBbox().getH()-MAX_STATE_AND_INFO_HEIGHT/2);
373
374        float parent_y_up = this.glyph.getBbox().getY()-INFO_BOUND.height/2;
375        float parent_y_bot = this.glyph.getBbox().getY()+this.glyph.getBbox().getH()-INFO_BOUND.height/2;;
376        float parent_x_up = this.glyph.getBbox().getX();
377        float parentWidth = this.glyph.getBbox().getW();
378        String parentID = this.glyph.getId();
379
380        int maxNumberOfStates =   stateGlyphs.size();
381        int maxNumberOfInfos =  infoGlyphs.size();
382        if (maxNumberOfInfos + maxNumberOfStates > 4)
383        {
384            if(maxNumberOfInfos < 2 || maxNumberOfStates < 2)
385            {
386                int minimum = Math.min(maxNumberOfInfos,maxNumberOfStates);
387
388                if (maxNumberOfInfos - minimum == 0)
389                {
390                    maxNumberOfInfos = minimum;
391                    maxNumberOfStates = maxNumberOfStates - maxNumberOfInfos;
392                }
393                else
394                {
395                    maxNumberOfStates = minimum;
396                    maxNumberOfInfos = maxNumberOfInfos - maxNumberOfStates;
397                }
398            }
399        }
400        //else normal placement 
401        
402        //Some variables for placement of state and info boxes
403        int totalNumberOfStateAndInfos = maxNumberOfInfos+maxNumberOfStates;
404        Glyph tmpglyph = null;
405        int infoIndex = 0;
406        int stateIndex = 0;
407        int topUsedWidth = 0;
408        int bottomUsedWidth = 0;
409        int additionalStateInfoOffset = 0;
410        boolean placeTopFlag = true;
411
412        //Place state and info glyphs
413        for (int i = 0; i < totalNumberOfStateAndInfos; i++)
414        {
415            //Adjust offsets between state and info boxes
416            int offsetMultiplier = i % 4 <= 1 ? 1 : 2;
417
418            if (maxNumberOfInfos - i > 0 && infoGlyphs.size() > 0)
419                tmpglyph =  infoGlyphs.get(infoIndex++);
420            else if (stateGlyphs.size() > 0)
421                tmpglyph = stateGlyphs.get(stateIndex++);
422
423            //Top placement
424            if (placeTopFlag)
425            {
426                if( totalNumberOfStateAndInfos <= 2 )
427                {
428                    tmpglyph.getBbox().setX(parent_x_up+parentWidth/2-tmpglyph.getBbox().getW()/2);
429                    tmpglyph.getBbox().setY(parent_y_up - additionalStateInfoOffset);
430                    //set dummy id
431                    tmpglyph.setId(parentID + ".info." + (i+1) );
432                }
433                else
434                {
435                    //set dummy id
436                    tmpglyph.setId(parentID + ".info." + (i+1) );
437
438                    tmpglyph.getBbox().setX(parent_x_up + offsetMultiplier * OFFSET_BTW_INFO_GLYPHS + topUsedWidth);
439                    tmpglyph.getBbox().setY(parent_y_up - additionalStateInfoOffset);
440                }
441
442                topUsedWidth += tmpglyph.getBbox().getW();
443            }
444            //bottom placement
445            else
446            {
447                if(( i % 2 == 1 ) && ( totalNumberOfStateAndInfos == 2 || totalNumberOfStateAndInfos == 3 ))
448                {
449                    tmpglyph.getBbox().setX(parent_x_up+parentWidth/2-tmpglyph.getBbox().getW()/2);
450                    tmpglyph.getBbox().setY(parent_y_bot + additionalStateInfoOffset);
451                    //set dummy id
452                    tmpglyph.setId(parentID + ".state." + (i+1) );
453                }
454                else
455                {
456                    //set dummy id
457                    tmpglyph.setId(parentID + ".state." + (i+1) );
458                    tmpglyph.getBbox().setX(parent_x_up+ offsetMultiplier *OFFSET_BTW_INFO_GLYPHS + bottomUsedWidth);
459                    tmpglyph.getBbox().setY(parent_y_bot + additionalStateInfoOffset);
460                }
461
462                bottomUsedWidth += tmpglyph.getBbox().getW();
463            }
464
465            // Alternate between placing top and bottom
466            placeTopFlag = !placeTopFlag;
467
468            //More than 4 state and info glyps are placed with an offset on top of previously placed state and info boxes
469            //This can be easily filtered out on the application side where the resultant graphs of this layout is used.
470            if ( ( i + 1 ) % 4 == 0 )
471                additionalStateInfoOffset += NON_SUPPORTED_GLYPH_OFFSET;
472        }
473
474    }
