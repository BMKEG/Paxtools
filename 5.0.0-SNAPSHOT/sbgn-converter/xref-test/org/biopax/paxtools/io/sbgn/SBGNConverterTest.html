31  	@Test
32  	public void testSBGNConversion() throws JAXBException, IOException, SAXException
33  	{
34  		String input = "/AR-TP53";
35  		InputStream in = getClass().getResourceAsStream(input + ".owl");
36  		Model level3 = handler.convertFromOWL(in);
37  
38  		Set<String> blacklist = new HashSet<String>(Arrays.asList(
39  			"http://pid.nci.nih.gov/biopaxpid_685", "http://pid.nci.nih.gov/biopaxpid_678",
40  			"http://pid.nci.nih.gov/biopaxpid_3119", "http://pid.nci.nih.gov/biopaxpid_3114"));
41  
42  		System.out.println("level3.getObjects().size() = " + level3.getObjects().size());
43  
44  		String out = "target/" + input + ".sbgn";
45  
46  		L3ToSBGNPDConverter conv = new L3ToSBGNPDConverter(
47  			new ListUbiqueDetector(blacklist), null, true);
48  
49  		conv.writeSBGN(level3, out);
50  
51  		File outFile = new File(out);
52  		boolean isValid = SbgnUtil.isValid(outFile);
53  
54  		if (isValid)
55  			System.out.println ("Validation succeeded");
56  		else
57  			System.out.println ("Validation failed");
58  
59  		JAXBContext context = JAXBContext.newInstance("org.sbgn.bindings");
60  		Unmarshaller unmarshaller = context.createUnmarshaller();
61  
62  		// Now read from "f" and put the result in "sbgn"
63  		Sbgn result = (Sbgn)unmarshaller.unmarshal (outFile);
64  
65  		// Assert that the sbgn result contains glyphs
66  		assertTrue(!result.getMap().getGlyph().isEmpty());
67  
68  		// Assert that compartments do not contain members inside
69  		for (Glyph g : result.getMap().getGlyph())
70  		{
71  			if (g.getClazz().equals("compartment"))
72  			{
73  				assertTrue(g.getGlyph().isEmpty());
74  			}
75  		}
76  
77  		// Assert that the id mapping is not empty.
78  		assertFalse(conv.getSbgn2BPMap().isEmpty());
79  
80  		for (Set<String> set : conv.getSbgn2BPMap().values())
81  		{
82  			assertFalse(set.isEmpty());
83  		}
84  	}

94  
95  		String out = "target/" + input + ".sbgn";
96  
97  		L3ToSBGNPDConverter conv = new L3ToSBGNPDConverter(null, null, true);
98  		conv.setFlattenComplexContent(false);
99  
100 		conv.writeSBGN(level3, out);
101 
102 		File outFile = new File(out);
103 		System.out.println("outFile.getPath() = " + outFile.getAbsolutePath());
104 		boolean isValid = SbgnUtil.isValid(outFile);
105 
106 		if (isValid)
107 			System.out.println ("Validation succeeded");
108 		else
109 			System.out.println ("Validation failed");
110 
111 		JAXBContext context = JAXBContext.newInstance("org.sbgn.bindings");
112 		Unmarshaller unmarshaller = context.createUnmarshaller();
113 
114 		// Now read from "f" and put the result in "sbgn"
115 		Sbgn result = (Sbgn)unmarshaller.unmarshal (outFile);
116 
117 		boolean hasNestedComplex = false;
118 		for (Glyph g : result.getMap().getGlyph())
119 		{
120 			if (g.getClazz().equals("complex"))
121 			{
122 				for (Glyph mem : g.getGlyph())
123 				{
124 					if (mem.getClazz().equals("complex"))
125 					{
126 						hasNestedComplex = true;
127 						break;
128 					}
129 				}
130 				if (hasNestedComplex) break;
131 			}
132 		}
133 
134 		assertTrue(hasNestedComplex);
135 	}
