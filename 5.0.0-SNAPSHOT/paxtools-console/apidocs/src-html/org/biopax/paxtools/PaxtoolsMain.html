056    public static void toGSEA(String[] argv) throws IOException {
057                //argv[0] is the command name ('toGsea')
058                boolean crossSpecies = false; //cross-check is enabled (i.e., no mixing different species IDs in one row)
059                boolean subPathways = false; //no sub-pathways (i.e., going into sub-pathways is not enabled)
060                boolean notPathways = false;
061                Set<String> organisms = new HashSet<String>();
062
063                if (argv.length < 4)
064                        throw new IllegalArgumentException("Not enough arguments: " + argv);
065
066                if (argv.length > 4) {
067                        for (int i = 4; i < argv.length; i++) {
068                                if("-crossSpecies".equalsIgnoreCase(argv[i])) {
069                                        crossSpecies = true;
070                                }
071                                else if("-subPathways".equalsIgnoreCase(argv[i])) {
072                                        subPathways = true;
073                                }
074                                else if("-notPathway".equalsIgnoreCase(argv[i])) {
075                                        notPathways = true;
076                                }
077                                else if(argv[i].startsWith("organisms=")) {
078                                        for(String o : argv[i].substring(10).split(",")) {
079                                                organisms.add(o.trim().toLowerCase());
080                                        }
081                                }
082                        }
083                }
084
085                // The Constructor args: GSEAConverter(idTypeNameOrPrefix, crossSpeciesCheckEnabled?, skipSubPathways?)
086                GSEAConverter gseaConverter = new GSEAConverter(argv[3], !crossSpecies, !subPathways);
087                gseaConverter.setSkipOutsidePathways(!notPathways);
088                gseaConverter.setAllowedOrganisms(organisms);//if organisms is empty then all species are allowed (no filtering)
089                gseaConverter.writeToGSEA(io.convertFromOWL(getInputStream(argv[1])), new FileOutputStream(argv[2]));
090    }

092    public static void getNeighbors(String[] argv) throws IOException
093    {
094        // set strings vars
095        String in = argv[1];
096        String[] ids = argv[2].split(",");
097        String out = argv[3];
098
099        // read BioPAX from the file
100        Model model = io.convertFromOWL(getInputStream(in));
101
102        // get elements (entities)
103        Set<BioPAXElement> elements = new HashSet<BioPAXElement>();
104        for (Object id : ids) {
105            BioPAXElement e = model.getByID(id.toString());
106            if (e != null && (e instanceof Entity || e instanceof entity)) {
107                elements.add(e);
108            } else {
109                log.warn("Source element not found: " + id);
110            }
111        }
112
113        // execute the 'nearest neighborhood' query
114        Set<BioPAXElement> result = QueryExecuter
115                        .runNeighborhood(elements, model, 1, Direction.BOTHSTREAM);
116
117        // auto-complete/clone the results in a new model
118        // (this also cuts some less important edges, right?..)
119        Completer c = new Completer(io.getEditorMap());
120        result = c.complete(result, model);
121        Cloner cln = new Cloner(io.getEditorMap(), io.getFactory());
122        model = cln.clone(model, result); // new sub-model
123
124        if (model != null) {
125            log.info("Elements in the result model: " + model.getObjects().size());
126            // export to OWL
127            io.convertToOWL(model, new FileOutputStream(out));
128        } else {
129            log.error("NULL model returned.");
130        }
131    }

147    /*
148     * Detects and converts a BioPAX Level 1, 2,
149     * or PSI-MI/MITAB model to BioPAX Level3..
150     */
151    public static void toLevel3(String[] argv) throws IOException {
152        final String input = argv[1];
153        final String output = argv[2];
154        InputStream is = getInputStream(input);
155        FileOutputStream os = new FileOutputStream(output);
156        
157        boolean forcePsiInteractionToComplex = false;
158                if(argv.length > 3) {
159                        for(int i=3; i<argv.length; i++) {
160                                String param = argv[i];
161                                if (param.equalsIgnoreCase("-psimiToComplexes")) {
162                                        forcePsiInteractionToComplex = true;
163                                }
164                        }
165                }
166
167                Type type = detect(input);
168
169        try {
170                switch (type) {
171                                case BIOPAX:
172                                        Model model = io.convertFromOWL(is);
173                                        model = (new LevelUpgrader()).filter(model);
174                                        if (model != null) {
175                                                io.setFactory(model.getLevel().getDefaultFactory());
176                                                io.convertToOWL(model, os); //os is closed already
177                                        }
178                                        break;
179
180                                case PSIMI:
181                                        PsiToBiopax3Converter psimiConverter = new PsiToBiopax3Converter();
182                                        psimiConverter.convert(is, os, forcePsiInteractionToComplex);
183                                        os.close();
184                                        break;
185
186                                default: //MITAB
187                                        psimiConverter = new PsiToBiopax3Converter();
188                                        psimiConverter.convertTab(is, os, forcePsiInteractionToComplex);
189                                        os.close();
190                                        break;
191                }
192
193        } catch (Exception e) {
194                throw new RuntimeException("Failed to convert " +
195                                input + "to BioPAX L3", e);
196        }
197    }

205    //read a few lines to detect it's a BioPAX vs. PSI-MI vs. PSI-MITAB data.
206        private static Type detect(String input) {              
207                StringBuilder sb = new StringBuilder();
208                try {
209                        BufferedReader reader = new BufferedReader(new FileReader(input));
210                        int linesToCheck = 20;
211                        while (linesToCheck-- > 0) {
212                                sb.append(reader.readLine()).append('\n');
213                        }
214                        reader.close();
215                } catch (IOException e) {
216                        throw new RuntimeException(e);
217                }
218                
219                String buf = sb.toString();
220                if (buf.contains("<rdf:RDF") && buf.contains("http://www.biopax.org/release/biopax")) {
221                        return Type.BIOPAX;
222                }
223                else if (buf.contains("<entrySet") && buf.contains("http://psidev.sourceforge.net/mi/rel25/")) {
224                        return Type.PSIMI;
225                }
226                else
227                        return Type.PSIMITAB; //default/guess
228        }

245    /*
246     * Checks files by the online BioPAX validator using the built-in BioPAX validator client.
247     * 
248     * See about <a href="http://www.biopax.org/validator">BioPAX Validator Webservice</a>
249     */
250    public static void validate(String[] argv) throws IOException 
251    {
252        String input = argv[1];
253        String output = argv[2];
254        // default options
255        RetFormat outf = RetFormat.HTML;
256        boolean fix = false;
257        Integer maxErrs = null;
258        Behavior level = null; //will report both errors and warnings
259        String profile = null;
260        
261        // match optional args
262                for (int i = 3; i < argv.length; i++) {
263                        if ("html".equalsIgnoreCase(argv[i])) {
264                                outf = RetFormat.HTML;
265                        } else if ("xml".equalsIgnoreCase(argv[i])) {
266                                outf = RetFormat.XML;
267                        } else if ("biopax".equalsIgnoreCase(argv[i])) {
268                                outf = RetFormat.OWL;
269                        } else if ("auto-fix".equalsIgnoreCase(argv[i])) {
270                                fix = true;
271                        } else if ("only-errors".equalsIgnoreCase(argv[i])) {
272                                level = Behavior.ERROR;
273                        } else if ((argv[i]).toLowerCase().startsWith("maxerrors=")) {
274                                String num = argv[i].substring(10);
275                                maxErrs = Integer.valueOf(num);
276                        } else if ("notstrict".equalsIgnoreCase(argv[i])) {
277                                profile = "notstrict";
278                        }
279                }
280
281        Collection<File> files = new HashSet<File>();
282        File fileOrDir = new File(input);
283        if (!fileOrDir.canRead()) {
284            System.out.println("Cannot read " + input);
285        }
286
287        // collect files
288        if (fileOrDir.isDirectory()) {
289            // validate all the OWL files in the folder
290            FilenameFilter filter = new FilenameFilter() {
291                public boolean accept(File dir, String name) {
292                    return (name.endsWith(".owl"));
293                }
294            };
295            for (String s : fileOrDir.list(filter)) {
296                files.add(new File(fileOrDir.getCanonicalPath()
297                        + File.separator + s));
298            }
299        } else {
300            files.add(fileOrDir);
301        }
302
303        // upload and validate using the default URL:
304        // http://www.biopax.org/biopax-validator/check.html
305        OutputStream os = new FileOutputStream(output);
306        try {
307            if (!files.isEmpty()) {
308                BiopaxValidatorClient val = new BiopaxValidatorClient();
309                val.validate(fix, profile, outf, level, maxErrs, null, files.toArray(new File[]{}), os);
310            }
311        } catch (Exception ex) {
312            // fall-back: not using the remote validator; trying to read files
313            String msg = "Unable to check with the biopax-validator web service: \n " +
314                        ex.toString() + 
315                        "\n Fall-back: trying to parse the file(s) with paxtools " +
316                        "(up to the first syntax error in each file)...\n";
317            log.error(msg, ex);
318            os.write(msg.getBytes());
319
320            for (File f : files) {
321                try {
322                    Model m = io.convertFromOWL(getInputStream(f.getPath()));
323                    msg = "Model that contains "
324                            + m.getObjects().size()
325                            + " elements is successfully created from " 
326                            + f.getPath() 
327                            + " (check the console output for warnings).\n";
328                    os.write(msg.getBytes());
329                } catch (Exception e) {
330                    msg = "Error: " + e + 
331                                " in building a BioPAX Model from: " +
332                                f.getPath() + "\n";
333                    os.write(msg.getBytes());
334                    e.printStackTrace();
335                    log.error(msg);
336                }
337                os.flush();
338            }
339        }
340    }

342        // Exports a biopax model to SIF or/and customizable ExtendedSIF format
343        // (default classic SIF has only 3 data columns; ExtendedSIF and customized
344        // can have more colums plus nodes description at the bottom, after a blank line).
345    public static void toSifnx(String[] argv) throws IOException
346        {
347                boolean extended = false; //if it stays 'false', then andSif==true will be in effect automatically
348                boolean andSif = false; //if extended==false, SIF will be generated as it would be andSif==true
349                boolean mergeInteractions = true;
350                boolean useNameIfNoId = false;
351                final Collection<SIFEnum> include = new HashSet<SIFEnum>();
352                final Collection<SIFEnum> exclude = new HashSet<SIFEnum>();
353
354                final ConfigurableIDFetcher idFetcher = new ConfigurableIDFetcher();
355                final List<String> customFieldList = new ArrayList<String>(); //there may be custom field names (SIF mediators)
356
357                //process arguments
358                if(argv.length > 3) {
359                        for(int i=3; i<argv.length; i++) {
360                                String param = argv[i];
361                                if (param.startsWith("seqDb=")) {
362                                        //remove the 'seqDb=' and split comma-sep. values (a single val. no comma is gonna be fine too)
363                                        for (String db : param.substring(6).split(","))
364                                                idFetcher.seqDbStartsWithOrEquals(db);
365                                } else if (param.startsWith("chemDb=")) {
366                                        for (String db : param.substring(7).split(","))
367                                                idFetcher.chemDbStartsWithOrEquals(db);
368                                } else if (param.equalsIgnoreCase("-andSif")) {
369                                        andSif = true;
370                                } else if (param.equalsIgnoreCase("-extended")) {
371                                        extended = true;
372                                } else if (param.equalsIgnoreCase("-dontMergeInteractions")) {
373                                        mergeInteractions = false;
374                                } else if (param.equalsIgnoreCase("-useNameIfNoId")) {
375                                        useNameIfNoId = true;
376                                } else if (param.startsWith("include=")) {
377                                        for (String t : param.substring(8).split(","))
378                                                include.add(SIFEnum.valueOf(t.toUpperCase()));
379                                } else if (param.startsWith("exclude=")) {
380                                        for (String t : param.substring(8).split(","))
381                                                exclude.add(SIFEnum.valueOf(t.toUpperCase())); //e.g. NEIGHBOR_OF
382                                } else {
383                                        OutputColumn.Type type = OutputColumn.Type.getType(param.toUpperCase());
384                                        if ((type != null && type != OutputColumn.Type.CUSTOM) || param.contains("/"))
385                                        {
386                                                if(!param.contains("/"))
387                                                        customFieldList.add(param.toUpperCase());
388                                                else
389                                                        customFieldList.add(param);
390                                        }
391                                }
392                        }
393                }
394
395                //fall back to defaults when no ID types were provided
396                if(idFetcher.getChemDbStartsWithOrEquals().isEmpty()) {
397                        idFetcher.chemDbStartsWithOrEquals("chebi");
398                }
399                if(idFetcher.getSeqDbStartsWithOrEquals().isEmpty()) {
400                        idFetcher.chemDbStartsWithOrEquals("hgnc");
401                }
402                idFetcher.useNameWhenNoDbMatch(useNameIfNoId);
403
404                Model model = getModel(io, argv[1]);
405
406                if(mergeInteractions)
407                        ModelUtils.mergeEquivalentInteractions(model);
408
409
410                //Create a new SIF searcher:
411                //set SIF miners to use (default is to use all types, given no include/exclude args provided)
412                final Collection<SIFEnum> sifTypes = (include.isEmpty())
413                                ? new HashSet<SIFEnum>(Arrays.asList(SIFEnum.values())) : include;
414                for(SIFType t : exclude) {
415                        sifTypes.remove(t); //remove if exists, otherwise - ignore
416                }
417                SIFSearcher searcher = new SIFSearcher(idFetcher, sifTypes.toArray(new SIFEnum[]{}));
418                log.info("toSIF: using SIFTypes: " + sifTypes);
419                //load and set blacklist.txt file if exists
420                File blacklistFile = new File("blacklist.txt");
421                if(blacklistFile.exists()) {
422                        log.info("toSIF: using blacklist.txt from current directory");
423                        searcher.setBlacklist(new Blacklist(new FileInputStream(blacklistFile)));
424                } else {
425                        log.info("toSIF: not blacklisting ubiquitous molecules (no blacklist.txt found)");
426                }
427
428                File outputFile = new File(argv[2]);
429                OutputStream outputStream = new FileOutputStream(outputFile);
430
431                if (extended) {
432                        //using built-in PC EXTENDED_BINARY_SIF format (customFieldList parameter is ignored)
433                        Set<SIFInteraction> binaryInts = searcher.searchSIF(model);
434                        ExtendedSIFWriter.write(binaryInts, outputStream);
435                }
436                else if (customFieldList.isEmpty()) {
437                        searcher.searchSIF(model, outputStream); //classic SIF
438                }
439                else {
440                // not really SIF format (this is useful but extra columns sometimes cannot be parsed correctly, by e.g. Cytoscape)
441                        log.info("toSIF: using custom fields (extra columns): " + customFieldList);
442                        searcher.searchSIF(model, outputStream, new CustomFormat(customFieldList.toArray(new String[]{})));
443                }
444                //outputStream is closed at this point (inside searchSIF(..) or write(..) method)
445
446                if(extended && andSif) {
447                        //convert the file into classic thee-column SIF format
448                        sifnxToSif(outputFile.getPath(), outputFile.getPath()+".sif");
449                }
450
451                log.info("toSIF: done.");
452    }

454        public static void sifnxToSif(String inputFile, String outputFile) throws IOException {
455                BufferedReader reader = new BufferedReader(new InputStreamReader(getInputStream(inputFile)));
456                OutputStreamWriter writer = new OutputStreamWriter(new FileOutputStream(outputFile));
457                //skip the first line (headers)
458                if(reader.ready()) reader.readLine();
459                while(reader.ready()) {
460                        String line = reader.readLine();
461                        //stop at the first blank line (because next come nodes with attributes)
462                        if(line==null || line.isEmpty())
463                                break;
464                        //keep only the first three columns (otherwise, it's not gonna be SIF format)
465                        writer.write(StringUtils.join(Arrays.copyOfRange(line.split("\t", 4), 0, 3), '\t') + '\n');
466                }
467                reader.close();
468                writer.close();
469        }

536        public static void summarize(Model model, PrintStream out) throws IOException {
537                if (out == null) out = System.out;
538
539        HashMap<String, Integer> hm = new HashMap<String, Integer>();
540
541                final SimpleEditorMap em = SimpleEditorMap.get(model.getLevel());
542
543                for (Class<? extends BioPAXElement> clazz : sortToName(em.getKnownSubClassesOf(BioPAXElement.class)))
544                {
545                        Set<? extends BioPAXElement> set = model.getObjects(clazz);
546                        int initialSize = set.size();
547                        set = filterToExactClass(set, clazz);
548                        String s = clazz.getSimpleName() + " = " + set.size();
549                        if (initialSize != set.size()) 
550                                s += " (and " + (initialSize - set.size()) + " children)";
551                        out.println(s);
552
553                        Set<PropertyEditor> editors = em.getEditorsOf(clazz);
554                        for (PropertyEditor editor : editors)
555                        {
556                                Method getMethod = editor.getGetMethod();
557                                Class<?> returnType = getMethod.getReturnType();
558
559                                Map<Object, Integer> cnt = new HashMap<Object, Integer>();
560
561                                if (returnType.isEnum() ||
562                                        implementsInterface(returnType, ControlledVocabulary.class))
563                                {
564                                        for (BioPAXElement ele : set)
565                                        {
566                                                Set<?> values = editor.getValueFromBean(ele);
567                                                if (values.isEmpty())
568                                                {
569                                                        increaseCnt(cnt, NULL);
570                                                }
571                                                else
572                                                {
573                                                        increaseCnt(cnt, values.iterator().next());
574                                                }
575                                        }
576                                }
577                                else if (returnType.equals(Set.class) &&
578                                        implementsInterface(editor.getRange(), ControlledVocabulary.class))
579                                {
580                                        for (BioPAXElement ele : set)
581                                        {
582                                                Set<?> values = editor.getValueFromBean(ele);
583                                                if (values.isEmpty())
584                                                {
585                                                        increaseCnt(cnt, EMPTY);
586                                                }
587                                                for (Object val : values)
588                                                {
589                                                        increaseCnt(cnt, val);
590                                                }
591                                        }
592                                }
593
594                                if (!cnt.isEmpty())
595                                {
596                                        String name = "-" 
597                                                + (returnType.equals(Set.class) ? editor.getRange().getSimpleName() : returnType.getSimpleName());
598
599                                        out.print("\t" + name + ":");
600                                        for (Object key : getOrdering(cnt))
601                                        {
602                                                out.print("\t" + key + " = " + cnt.get(key));
603                                        }
604                                        out.println();
605                                }
606                        }
607                }
608
609                out.println("\nOther property counts\n");
610                String[] props = (model.getLevel() == BioPAXLevel.L3)
611                        ? new String[]{"UnificationXref/db","RelationshipXref/db"}
612                        : new String[]{"unificationXref/DB","relationshipXref/DB"};
613                for (String prop : props)
614                {
615                        Map<Object, Integer> cnt = new HashMap<Object, Integer>();
616                        List<String> valList = new ArrayList<String>();
617                        PathAccessor acc = new PathAccessor(prop, model.getLevel());
618                        
619                        boolean isString = false;
620                        
621                        for (Object o : acc.getValueFromModel(model))
622                        {
623                                if (o instanceof String) isString = true;
624                                
625                                String s = o.toString();
626                                valList.add(s);
627                                if (!cnt.containsKey(s)) cnt.put(s, 1);
628                                else cnt.put(s, cnt.get(s) + 1);
629                        }
630
631                        out.println(prop + "\t(" + cnt.size() + " distinct values):");
632            hm.put(prop, cnt.size());
633
634                        // If the object is String, then all counts are 1, no need to print counts.
635                        if (isString)
636                        {
637                                Collections.sort(valList);
638                                for (String s : valList)
639                                {
640                                        out.print("\t" + s);
641                                }
642                        }
643                        else
644                        {
645                                for (Object key : getOrdering(cnt))
646                                {
647                                        out.print("\t" + key + " = " + cnt.get(key));
648                                }
649                        }
650                        out.println();
651                }
652
653                //Count simple PEs that have null entityReference
654                int speLackingEr = 0;
655                int genericSpeLackingEr = 0;
656                int speLackingErAndId = 0;
657                int protLackingErAndId = 0;
658                int molLackingErAndId = 0;
659                int naLackingErAndId = 0; //rem: na - nucleic acid
660                //a map that contains the no. SPEs that have null entityReference by data source
661                Map<String,Integer> numSpeLackErByProvider = new TreeMap<String, Integer>();
662                for(SimplePhysicalEntity spe : model.getObjects(SimplePhysicalEntity.class)) {
663                        if(spe.getEntityReference()==null) {
664                                speLackingEr++;
665                                if(!spe.getMemberPhysicalEntity().isEmpty())
666                                        genericSpeLackingEr++;
667
668                                String providers = spe.getDataSource().toString();
669                                Integer n = numSpeLackErByProvider.get(providers);
670                                n = (n==null) ? 1 : n + 1;
671                                numSpeLackErByProvider.put(providers, n);
672
673                                if(spe.getXref().isEmpty() ||
674                                        new ClassFilterSet<Xref,PublicationXref>(spe.getXref(), PublicationXref.class)
675                                                .size() == spe.getXref().size())
676                                {
677                                        speLackingErAndId++;
678                                        if(spe instanceof Protein)
679                                                protLackingErAndId++;
680                                        else if(spe instanceof SmallMolecule)
681                                                molLackingErAndId++;
682                                        else if(spe instanceof NucleicAcid)
683                                                naLackingErAndId++;
684                                }
685                        }
686                }
687
688                out.println("\n" + speLackingEr + " simple physical entities have NULL 'entityReference';\n");
689                out.println("\n\t-" + genericSpeLackingEr + " of which have member physical entities (are generic).\n");
690                out.println("\n\t- by data source:\n");
691                for(String key : numSpeLackErByProvider.keySet()) {
692                        out.println(String.format("\n\t\t-- %s -> %d\n", key, numSpeLackErByProvider.get(key)));
693                }
694                out.println("\n\t- " + speLackingErAndId + " neither have 'entityReference' nor xref/id (except publications):\n");
695                if(speLackingErAndId > 0) {
696                        out.println("\n\t\t-- proteins: " + protLackingErAndId + "\n");
697                        out.println("\n\t\t-- small molecules: " + molLackingErAndId + "\n");
698                        out.println("\n\t\t-- nucl. acids: " + naLackingErAndId + "\n");
699                }
700
701                int erLackingId = 0;
702                for(EntityReference er : model.getObjects(EntityReference.class)) {
703                        if(er.getMemberEntityReference().isEmpty() &&
704                                (er.getXref().isEmpty() || new ClassFilterSet<Xref,PublicationXref>(er.getXref(), PublicationXref.class)
705                                        .size() == er.getXref().size()))
706                        {
707                                erLackingId++;
708                        }
709                }
710                out.println("\n" + erLackingId + " non-generic entity references have no xref/id.\n");
711
712                //The number of sequence ERs (not generic), Genes, Pathways, where 'organism' property is empty -
713                int genesLackingOrganism = 0;
714                int pwLackingOrganism = 0;
715                int serLackingOrganism = 0;
716                int narLackingOrganism = 0;
717                for(BioPAXElement bpe : model.getObjects()) {
718                        if(bpe instanceof Gene && ((Gene) bpe).getOrganism()==null)
719                                ++genesLackingOrganism;
720                        else if(bpe instanceof Pathway && ((Pathway) bpe).getOrganism()==null)
721                                ++pwLackingOrganism;
722                        else if(bpe instanceof SequenceEntityReference && ((SequenceEntityReference)bpe).getOrganism() == null) {
723                                ++serLackingOrganism;
724                                if(bpe instanceof NucleicAcidReference)
725                                        ++narLackingOrganism;
726                        }
727                }
728                out.println(
729                        String.format(
730                                "\n%d Genes, %d Pathways, %d SequenceEntityReferences " +
731                                "(%d in NucleicAcidRef. and %d in PRs) have NULL 'organism'.\n",
732                                genesLackingOrganism, pwLackingOrganism, serLackingOrganism,
733                                narLackingOrganism, serLackingOrganism-narLackingOrganism
734                        )
735                );
736        }

738        private static List<Class<? extends BioPAXElement>> sortToName(Set<? extends Class<? extends BioPAXElement>>
739                        classes)
740        {
741                List<Class<? extends BioPAXElement>> list = new ArrayList<Class<? extends BioPAXElement>>(classes);
742
743                Collections.sort(
744                                list, new Comparator<Class<? extends  BioPAXElement>>()
745                {
746                        public int compare(Class<? extends BioPAXElement> clazz1, Class<? extends BioPAXElement> clazz2)
747                        {
748                                return clazz1.getName().substring(clazz1.getName().lastIndexOf(".")+1).compareTo(
749                                        clazz2.getName().substring(clazz2.getName().lastIndexOf(".")+1));
750                        }
751                });
752                return list;
753        }

755        private static List<Object> getOrdering(final Map<Object, Integer> map)
756        {
757                List<Object> list = new ArrayList<Object>(map.keySet());
758                Collections.sort(list, new Comparator<Object>()
759                {
760                        public int compare(Object key1, Object key2)
761                        {
762                                int cnt1 = map.get(key1);
763                                int cnt2 = map.get(key2);
764
765                                if (cnt1 == cnt2) return key1.toString().compareTo(key2.toString());
766                                else return cnt2 - cnt1;
767                        }
768                });
769                return list;
770        }

772        private static Set<BioPAXElement> filterToExactClass(Set<? extends BioPAXElement> classSet, Class<?> clazz)
773        {
774                Set<BioPAXElement> exact = new HashSet<BioPAXElement>();
775                for (BioPAXElement ele : classSet)
776                {
777                        if (ele.getModelInterface().equals(clazz)) exact.add(ele);
778                }
779                return exact;
