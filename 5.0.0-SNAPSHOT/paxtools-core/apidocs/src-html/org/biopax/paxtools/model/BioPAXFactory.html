001package org.biopax.paxtools.model;
002
003import org.biopax.paxtools.impl.BioPAXElementImpl;
004import org.biopax.paxtools.impl.ModelImpl;
005import org.slf4j.Logger;
006import org.slf4j.LoggerFactory;
007
008import java.lang.reflect.Constructor;
009import java.lang.reflect.Method;
010import java.lang.reflect.Modifier;
011
012/**
013 * Abstract factory class for instantiating BioPAX classes. Different implementations of BioPAX model should also
014 * implement their own factory
015 */
016
017public abstract class BioPAXFactory
018{
019    private static final Logger log = LoggerFactory.getLogger(BioPAXFactory.class);
020        
021    private final Method setUriMethod;
022    
023    /**
024     * Protected Constructor without parameters.
025     */
026    protected BioPAXFactory() {
027        try {
028                        setUriMethod = BioPAXElementImpl.class.getDeclaredMethod("setUri", String.class);
029                        setUriMethod.setAccessible(true);
030                } catch (Throwable e) {
031                        throw new RuntimeException("BioPAXFactory Constructor failed", e);
032                } 
033        }
034
035    /**
036     * Gets the level.
037     * 
038     * @return the biopax level
039     */
040    public abstract BioPAXLevel getLevel();
041
042    
043    public BioPAXElement create(String localName, String uri)
044    {
045        Class<? extends BioPAXElement> type = getLevel().getInterfaceForName(localName);
046        return create(type, uri);
047    }
048
049    
050    /**
051     * Universal method that creates a new BioPAX object.
052     * (works with non-public, other package, implementations)
053         *
054         * @param <T> type
055         * @param aClass the class that corresponds to the BioPAX type
056         * @param uri absolute URI of the new BioPAX object
057         * @return new BioPAX object
058     */
059        public <T extends BioPAXElement> T create(Class<T> aClass, String uri) {
060                T bpe = null;
061
062                // create a new instance of the BioPAX type
063                try {
064                        Class<T> t = getImplClass(aClass);
065                        if(t != null) {
066                                Constructor<T> c = t.getDeclaredConstructor();
067                                c.setAccessible(true);
068                                bpe = (T) c.newInstance();
069                                setUriMethod.invoke(bpe, uri);
070                        } else {
071                                log.error("Could not find a class implementing " + aClass);
072                                return null;
073                        }
074                } catch (Exception e) {
075                        log.error("Could not instantiate BioPAX Type: " + aClass 
076                                        + "; URI: " + uri, e);
077                } 
078
079                return bpe;
080        }
081
082    
083        /**
084         * Maps a BioPAX type (model interface) to the
085         * full-qualified class name of an implementing class.
086         * BioPAX factories have to implement this method.
087         * 
088         * @param aClass BioPAX type (model interface)
089         * @return full class name
090         */
091    public abstract String mapClassName(Class<? extends BioPAXElement> aClass);
092
093    
094    /**
095     * Checks whether objects of a BioPAX model interface
096     * can be created (some types are not official BioPAX 
097     * types, abstract classes).
098     * 
099     * @param aClass BioPAX interface class
100     * @return whether this factory can create an instance of the type
101     */
102    public boolean canInstantiate(Class<? extends BioPAXElement> aClass) 
103    {
104        try {
105            String cname = mapClassName(aClass);
106                return !Modifier.isAbstract(Class.forName(cname).getModifiers());
107        } catch (ClassNotFoundException e)
108        {
109            return false;
110        } catch (Exception ex) {
111                log.error("Error in canInstantiate(" + aClass + ")", ex);
112                        return false;
113                }
114    }
115
116
117        /**
118         * Creates a new BioPAX model.
119         * @return BioPAX object model implementation
120         */
121        public Model createModel() {
122        return new ModelImpl(this);
123    }
124
125    
126    /**
127     * Get a concrete or abstract BioPAX type (not interface), 
128     * from org.biopax.paxtools.impl..*, i.e., one that has 
129     * persistence/search annotations, etc. This may be required for
130     * some DAO and web service controllers; it also returns such 
131     * abstract BioPAX "adapters" as XReferrableImpl, ProcessImpl, etc.
132     * 
133     * @param <T> BioPAX type/interface
134     * @param aModelInterfaceClass interface class for the BioPAX type
135     * @return concrete class that implements the BioPAX interface and can be created with this factory
136     */
137        public <T extends BioPAXElement> Class<T> getImplClass(
138                        Class<T> aModelInterfaceClass) 
139        {
140                Class<T> implClass = null;
141
142                if (aModelInterfaceClass.isInterface()) {
143                        String name = mapClassName(aModelInterfaceClass);
144                        try {
145                                implClass = (Class<T>) Class.forName(name);
146                        } catch (ClassNotFoundException e) {
147                                log.debug(String.format("getImplClass(%s), %s" , aModelInterfaceClass, e));
148                        }
149                }
150
151                return implClass;
152        }
153
154}
