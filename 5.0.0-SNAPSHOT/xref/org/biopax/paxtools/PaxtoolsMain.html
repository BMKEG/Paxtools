55      public static void toGSEA(String[] argv) throws IOException {
56  		//argv[0] is the command name ('toGsea')
57  		boolean crossSpecies = false; //cross-check is enabled (i.e., no mixing different species IDs in one row)
58  		boolean subPathways = false; //no sub-pathways (i.e., going into sub-pathways is not enabled)
59  		boolean notPathways = false;
60  		Set<String> organisms = new HashSet<String>();
61  
62  		if (argv.length < 4)
63  			throw new IllegalArgumentException("Not enough arguments: " + argv);
64  
65  		if (argv.length > 4) {
66  			for (int i = 4; i < argv.length; i++) {
67  				if("-crossSpecies".equalsIgnoreCase(argv[i])) {
68  					crossSpecies = true;
69  				}
70  				else if("-subPathways".equalsIgnoreCase(argv[i])) {
71  					subPathways = true;
72  				}
73  				else if("-notPathway".equalsIgnoreCase(argv[i])) {
74  					notPathways = true;
75  				}
76  				else if(argv[i].startsWith("organisms=")) {
77  					for(String o : argv[i].substring(10).split(",")) {
78  						organisms.add(o.trim().toLowerCase());
79  					}
80  				}
81  			}
82  		}
83  
84  		// The Constructor args: GSEAConverter(idTypeNameOrPrefix, crossSpeciesCheckEnabled?, skipSubPathways?)
85  		GSEAConverter gseaConverter = new GSEAConverter(argv[3], !crossSpecies, !subPathways);
86  		gseaConverter.setSkipOutsidePathways(!notPathways);
87  		gseaConverter.setAllowedOrganisms(organisms);//if organisms is empty then all species are allowed (no filtering)
88  		gseaConverter.writeToGSEA(io.convertFromOWL(getInputStream(argv[1])), new FileOutputStream(argv[2]));
89      }

91      public static void getNeighbors(String[] argv) throws IOException
92      {
93          // set strings vars
94          String in = argv[1];
95          String[] ids = argv[2].split(",");
96          String out = argv[3];
97  
98          // read BioPAX from the file
99          Model model = io.convertFromOWL(getInputStream(in));
100 
101         // get elements (entities)
102         Set<BioPAXElement> elements = new HashSet<BioPAXElement>();
103         for (Object id : ids) {
104             BioPAXElement e = model.getByID(id.toString());
105             if (e != null && (e instanceof Entity || e instanceof entity)) {
106                 elements.add(e);
107             } else {
108                 log.warn("Source element not found: " + id);
109             }
110         }
111 
112         // execute the 'nearest neighborhood' query
113         Set<BioPAXElement> result = QueryExecuter
114         		.runNeighborhood(elements, model, 1, Direction.BOTHSTREAM);
115 
116         // auto-complete/clone the results in a new model
117         // (this also cuts some less important edges, right?..)
118         Completer c = new Completer(io.getEditorMap());
119         result = c.complete(result, model);
120         Cloner cln = new Cloner(io.getEditorMap(), io.getFactory());
121         model = cln.clone(model, result); // new sub-model
122 
123         if (model != null) {
124             log.info("Elements in the result model: " + model.getObjects().size());
125             // export to OWL
126             io.convertToOWL(model, new FileOutputStream(out));
127         } else {
128             log.error("NULL model returned.");
129         }
130     }

132     public static void fetch(String[] argv) throws IOException {
133         // set strings vars
134         String in = argv[1];
135         String out = argv[2];
136 		String[] uris = new String[0]; //empty means - all
137 		boolean absoluteUris = false;
138 		if(argv.length > 3) {
139 			for(int i=3; i<argv.length; i++) {
140 				String param = argv[i];
141 				if (param.startsWith("uris=")) {
142 					uris = param.substring(5).split(",");
143 				}
144 				else if(param.startsWith("-absolute")) {
145 					absoluteUris = true;
146 				}
147 			}
148 		}
149 
150 		// import the model
151 		log.info("Loading the BioPAX model from " + in);
152         Model model = io.convertFromOWL(getInputStream(in));
153 		log.info("Successfully loaded the BioPAX model. Writing to the output: " + out);
154 
155 		// extract and save the (sub-)model
156 		SimpleIOHandler biopaxWriter = new SimpleIOHandler(model.getLevel());
157 		biopaxWriter.absoluteUris(absoluteUris);
158 		if(uris.length > 0)
159 			biopaxWriter.convertToOWL(model, new FileOutputStream(out), uris);
160 		else
161 			biopaxWriter.convertToOWL(model, new FileOutputStream(out));
162 
163 		log.info("Done.");
164     }

167     /*
168      * Detects and converts a BioPAX Level 1, 2,
169      * or PSI-MI/MITAB model to BioPAX Level3..
170      */
171     public static void toLevel3(String[] argv) throws IOException {
172     	final String input = argv[1];
173     	final String output = argv[2];
174     	InputStream is = getInputStream(input);
175     	FileOutputStream os = new FileOutputStream(output);
176     	
177     	boolean forcePsiInteractionToComplex = false;
178 		if(argv.length > 3) {
179 			for(int i=3; i<argv.length; i++) {
180 				String param = argv[i];
181 				if (param.equalsIgnoreCase("-psimiToComplexes")) {
182 					forcePsiInteractionToComplex = true;
183 				}
184 			}
185 		}
186 
187 		Type type = detect(input);
188 
189     	try {
190     		switch (type) {
191 				case BIOPAX:
192 					Model model = io.convertFromOWL(is);
193 					model = (new LevelUpgrader()).filter(model);
194 					if (model != null) {
195 						io.setFactory(model.getLevel().getDefaultFactory());
196 						io.convertToOWL(model, os); //os is closed already
197 					}
198 					break;
199 
200 				case PSIMI:
201 					PsiToBiopax3Converter psimiConverter = new PsiToBiopax3Converter();
202 					psimiConverter.convert(is, os, forcePsiInteractionToComplex);
203 					os.close();
204 					break;
205 
206 				default: //MITAB
207 					psimiConverter = new PsiToBiopax3Converter();
208 					psimiConverter.convertTab(is, os, forcePsiInteractionToComplex);
209 					os.close();
210 					break;
211     		}
212 
213     	} catch (Exception e) {
214     		throw new RuntimeException("Failed to convert " +
215     				input + "to BioPAX L3", e);
216     	}
217     }

225     //read a few lines to detect it's a BioPAX vs. PSI-MI vs. PSI-MITAB data.
226 	private static Type detect(String input) {		
227 		StringBuilder sb = new StringBuilder();
228 		try {
229 			BufferedReader reader = new BufferedReader(new FileReader(input));
230 			int linesToCheck = 20;
231 			while (linesToCheck-- > 0) {
232 				sb.append(reader.readLine()).append('\n');
233 			}
234 			reader.close();
235 		} catch (IOException e) {
236 			throw new RuntimeException(e);
237 		}
238 		
239 		String buf = sb.toString();
240 		if (buf.contains("<rdf:RDF") && buf.contains("http://www.biopax.org/release/biopax")) {
241 			return Type.BIOPAX;
242 		}
243 		else if (buf.contains("<entrySet") && buf.contains("http://psidev.sourceforge.net/mi/rel25/")) {
244 			return Type.PSIMI;
245 		}
246 		else
247 			return Type.PSIMITAB; //default/guess
248 	}

266      * Checks files by the online BioPAX validator using the built-in BioPAX validator client.
267      * 
268      * See about <a href="http://www.biopax.org/validator">BioPAX Validator Webservice</a>
269      */
270     public static void validate(String[] argv) throws IOException 
271     {
272         String input = argv[1];
273         String output = argv[2];
274         // default options
275         RetFormat outf = RetFormat.HTML;
276         boolean fix = false;
277         Integer maxErrs = null;
278         Behavior level = null; //will report both errors and warnings
279         String profile = null;
280         
281         // match optional args
282 		for (int i = 3; i < argv.length; i++) {
283 			if ("html".equalsIgnoreCase(argv[i])) {
284 				outf = RetFormat.HTML;
285 			} else if ("xml".equalsIgnoreCase(argv[i])) {
286 				outf = RetFormat.XML;
287 			} else if ("biopax".equalsIgnoreCase(argv[i])) {
288 				outf = RetFormat.OWL;
289 			} else if ("auto-fix".equalsIgnoreCase(argv[i])) {
290 				fix = true;
291 			} else if ("only-errors".equalsIgnoreCase(argv[i])) {
292 				level = Behavior.ERROR;
293 			} else if ((argv[i]).toLowerCase().startsWith("maxerrors=")) {
294 				String num = argv[i].substring(10);
295 				maxErrs = Integer.valueOf(num);
296 			} else if ("notstrict".equalsIgnoreCase(argv[i])) {
297 				profile = "notstrict";
298 			}
299 		}
300 
301         Collection<File> files = new HashSet<File>();
302         File fileOrDir = new File(input);
303         if (!fileOrDir.canRead()) {
304             System.out.println("Cannot read " + input);
305         }
306 
307         // collect files
308         if (fileOrDir.isDirectory()) {
309             // validate all the OWL files in the folder
310             FilenameFilter filter = new FilenameFilter() {
311                 public boolean accept(File dir, String name) {
312                     return (name.endsWith(".owl"));
313                 }
314             };
315             for (String s : fileOrDir.list(filter)) {
316                 files.add(new File(fileOrDir.getCanonicalPath()
317                         + File.separator + s));
318             }
319         } else {
320             files.add(fileOrDir);
321         }
322 
323         // upload and validate using the default URL:
324         // http://www.biopax.org/biopax-validator/check.html
325         OutputStream os = new FileOutputStream(output);
326         try {
327             if (!files.isEmpty()) {
328                 BiopaxValidatorClient val = new BiopaxValidatorClient();
329                 val.validate(fix, profile, outf, level, maxErrs, null, files.toArray(new File[]{}), os);
330             }
331         } catch (Exception ex) {
332             // fall-back: not using the remote validator; trying to read files
333             String msg = "Unable to check with the biopax-validator web service: \n " +
334             		ex.toString() + 
335             		"\n Fall-back: trying to parse the file(s) with paxtools " +
336             		"(up to the first syntax error in each file)...\n";
337             log.error(msg, ex);
338             os.write(msg.getBytes());
339 
340             for (File f : files) {
341                 try {
342                     Model m = io.convertFromOWL(getInputStream(f.getPath()));
343                     msg = "Model that contains "
344                             + m.getObjects().size()
345                             + " elements is successfully created from " 
346                             + f.getPath() 
347                             + " (check the console output for warnings).\n";
348                     os.write(msg.getBytes());
349                 } catch (Exception e) {
350                     msg = "Error: " + e + 
351                     		" in building a BioPAX Model from: " +
352                     		f.getPath() + "\n";
353                     os.write(msg.getBytes());
354                     e.printStackTrace();
355                     log.error(msg);
356                 }
357                 os.flush();
358             }
359         }
360     }

362 	// Exports a biopax model to SIF or/and customizable ExtendedSIF format
363 	// (default classic SIF has only 3 data columns; ExtendedSIF and customized
364 	// can have more colums plus nodes description at the bottom, after a blank line).
365     public static void toSifnx(String[] argv) throws IOException
366 	{
367 		boolean extended = false; //if it stays 'false', then andSif==true will be in effect automatically
368 		boolean andSif = false; //if extended==false, SIF will be generated as it would be andSif==true
369 		boolean mergeInteractions = true;
370 		boolean useNameIfNoId = false;
371 		final Collection<SIFEnum> include = new HashSet<SIFEnum>();
372 		final Collection<SIFEnum> exclude = new HashSet<SIFEnum>();
373 
374 		final ConfigurableIDFetcher idFetcher = new ConfigurableIDFetcher();
375 		final List<String> customFieldList = new ArrayList<String>(); //there may be custom field names (SIF mediators)
376 
377 		//process arguments
378 		if(argv.length > 3) {
379 			for(int i=3; i<argv.length; i++) {
380 				String param = argv[i];
381 				if (param.startsWith("seqDb=")) {
382 					//remove the 'seqDb=' and split comma-sep. values (a single val. no comma is gonna be fine too)
383 					for (String db : param.substring(6).split(","))
384 						idFetcher.seqDbStartsWithOrEquals(db);
385 				} else if (param.startsWith("chemDb=")) {
386 					for (String db : param.substring(7).split(","))
387 						idFetcher.chemDbStartsWithOrEquals(db);
388 				} else if (param.equalsIgnoreCase("-andSif")) {
389 					andSif = true;
390 				} else if (param.equalsIgnoreCase("-extended")) {
391 					extended = true;
392 				} else if (param.equalsIgnoreCase("-dontMergeInteractions")) {
393 					mergeInteractions = false;
394 				} else if (param.equalsIgnoreCase("-useNameIfNoId")) {
395 					useNameIfNoId = true;
396 				} else if (param.startsWith("include=")) {
397 					for (String t : param.substring(8).split(","))
398 						include.add(SIFEnum.valueOf(t.toUpperCase()));
399 				} else if (param.startsWith("exclude=")) {
400 					for (String t : param.substring(8).split(","))
401 						exclude.add(SIFEnum.valueOf(t.toUpperCase())); //e.g. NEIGHBOR_OF
402 				} else {
403 					OutputColumn.Type type = OutputColumn.Type.getType(param.toUpperCase());
404 					if ((type != null && type != OutputColumn.Type.CUSTOM) || param.contains("/"))
405 					{
406 						if(!param.contains("/"))
407 							customFieldList.add(param.toUpperCase());
408 						else
409 							customFieldList.add(param);
410 					}
411 				}
412 			}
413 		}
414 
415 		//fall back to defaults when no ID types were provided
416 		if(idFetcher.getChemDbStartsWithOrEquals().isEmpty()) {
417 			idFetcher.chemDbStartsWithOrEquals("chebi");
418 		}
419 		if(idFetcher.getSeqDbStartsWithOrEquals().isEmpty()) {
420 			idFetcher.chemDbStartsWithOrEquals("hgnc");
421 		}
422 		idFetcher.useNameWhenNoDbMatch(useNameIfNoId);
423 
424 		Model model = getModel(io, argv[1]);
425 
426 		if(mergeInteractions)
427 			ModelUtils.mergeEquivalentInteractions(model);
428 
429 
430 		//Create a new SIF searcher:
431 		//set SIF miners to use (default is to use all types, given no include/exclude args provided)
432 		final Collection<SIFEnum> sifTypes = (include.isEmpty())
433 				? new HashSet<SIFEnum>(Arrays.asList(SIFEnum.values())) : include;
434 		for(SIFType t : exclude) {
435 			sifTypes.remove(t); //remove if exists, otherwise - ignore
436 		}
437 		SIFSearcher searcher = new SIFSearcher(idFetcher, sifTypes.toArray(new SIFEnum[]{}));
438 		log.info("toSIF: using SIFTypes: " + sifTypes);
439 		//load and set blacklist.txt file if exists
440 		File blacklistFile = new File("blacklist.txt");
441 		if(blacklistFile.exists()) {
442 			log.info("toSIF: using blacklist.txt from current directory");
443 			searcher.setBlacklist(new Blacklist(new FileInputStream(blacklistFile)));
444 		} else {
445 			log.info("toSIF: not blacklisting ubiquitous molecules (no blacklist.txt found)");
446 		}
447 
448 		File outputFile = new File(argv[2]);
449 		OutputStream outputStream = new FileOutputStream(outputFile);
450 
451 		if (extended) {
452 			//using built-in PC EXTENDED_BINARY_SIF format (customFieldList parameter is ignored)
453 			Set<SIFInteraction> binaryInts = searcher.searchSIF(model);
454 			ExtendedSIFWriter.write(binaryInts, outputStream);
455 		}
456 		else if (customFieldList.isEmpty()) {
457 			searcher.searchSIF(model, outputStream); //classic SIF
458 		}
459 		else {
460 		// not really SIF format (this is useful but extra columns sometimes cannot be parsed correctly, by e.g. Cytoscape)
461 			log.info("toSIF: using custom fields (extra columns): " + customFieldList);
462 			searcher.searchSIF(model, outputStream, new CustomFormat(customFieldList.toArray(new String[]{})));
463 		}
464 		//outputStream is closed at this point (inside searchSIF(..) or write(..) method)
465 
466 		if(extended && andSif) {
467 			//convert the file into classic thee-column SIF format
468 			sifnxToSif(outputFile.getPath(), outputFile.getPath()+".sif");
469 		}
470 
471 		log.info("toSIF: done.");
472     }

474 	public static void sifnxToSif(String inputFile, String outputFile) throws IOException {
475 		BufferedReader reader = new BufferedReader(new InputStreamReader(getInputStream(inputFile)));
476 		OutputStreamWriter writer = new OutputStreamWriter(new FileOutputStream(outputFile));
477 		//skip the first line (headers)
478 		if(reader.ready()) reader.readLine();
479 		while(reader.ready()) {
480 			String line = reader.readLine();
481 			//stop at the first blank line (because next come nodes with attributes)
482 			if(line==null || line.isEmpty())
483 				break;
484 			//keep only the first three columns (otherwise, it's not gonna be SIF format)
485 			writer.write(StringUtils.join(Arrays.copyOfRange(line.split("\t", 4), 0, 3), '\t') + '\n');
486 		}
487 		reader.close();
488 		writer.close();
489 	}

571 	private static void summarizePathways(Model model, PrintStream out) throws IOException {
572 		final PathAccessor directChildPathwaysAccessor = new PathAccessor("Pathway/pathwayComponent:Pathway");
573 		final PathAccessor pathwayComponentAccessor = new PathAccessor("Pathway/pathwayComponent");
574 		final PathAccessor pathwayOrderStepProcessAccessor = new PathAccessor("Pathway/pathwayOrder/stepProcess");
575 		//get all pathways sorted by name (or URI if no display name found)
576 		Set<Pathway> pathways = new TreeSet<Pathway>(new Comparator<Pathway>() {
577 			@Override
578 			public int compare(Pathway o1, Pathway o2) {
579 				return (o1.getDisplayName()!=null && o2.getDisplayName()!=null)

695 		out.println("\nNumber of SequenceEntityReferences (not generics) without any HGNC ID, by data source:");
696 		int totalPrs = 0;

702 			totalPrs += t;
703 			out.println(String.format("%s\t\t%d\t(%3.1f%%)", ds.getUri(), n, ((float)n)/t*100));
704 		}
705 		out.println(String.format("Total\t\t%d\t(%3.1f%%)", numPrsNoHgnc, ((float)numPrsNoHgnc)/totalPrs*100));
706 	}

1027 	private static List<Class<? extends BioPAXElement>> sortToName(Set<? extends Class<? extends BioPAXElement>>
1028 			classes)
1029 	{
1030 		List<Class<? extends BioPAXElement>> list = new ArrayList<Class<? extends BioPAXElement>>(classes);
1031 
1032 		Collections.sort(
1033 				list, new Comparator<Class<? extends  BioPAXElement>>()
1034 		{
1035 			public int compare(Class<? extends BioPAXElement> clazz1, Class<? extends BioPAXElement> clazz2)
1036 			{
1037 				return clazz1.getName().substring(clazz1.getName().lastIndexOf(".")+1).compareTo(
1038 					clazz2.getName().substring(clazz2.getName().lastIndexOf(".")+1));
1039 			}
1040 		});

1044 	private static List<Object> getOrdering(final Map<Object, Integer> map)
1045 	{
1046 		List<Object> list = new ArrayList<Object>(map.keySet());
1047 		Collections.sort(list, new Comparator<Object>()
1048 		{
1049 			public int compare(Object key1, Object key2)
1050 			{
1051 				int cnt1 = map.get(key1);
1052 				int cnt2 = map.get(key2);
1053 
1054 				if (cnt1 == cnt2) return key1.toString().compareTo(key2.toString());
1055 				else return cnt2 - cnt1;
1056 			}
1057 		});
1058 		return list;
1059 	}
