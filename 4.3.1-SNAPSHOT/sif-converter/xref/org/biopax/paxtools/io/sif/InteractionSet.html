3   import org.biopax.paxtools.util.AccessibleSet;
4   
5   /**
6    * This is a set of binary interactions. When a new interactions is added, if the pair already
7    * exists, only the mediators are updated with new information.
8    */
9   public class InteractionSet extends AccessibleSet<SimpleInteraction>
10  {
11  	/**
12  	 * If the interaction already exists, mediators are updated with merging.
13  	 * @param simpleInteraction interaction to add
14  	 * @return always true
15  	 */
16  	@Override public boolean add(SimpleInteraction simpleInteraction)
17  	{
18  
19  		SimpleInteraction existing = access(simpleInteraction);
20  		if (existing == null)
21  		{
22  			existing = simpleInteraction;
23  			super.add(existing);
24  		}
25  		else
26  		{
27  			existing.getMediators().addAll(simpleInteraction.getMediators());
28  		}
29  		return true;
30  	}
31  }
