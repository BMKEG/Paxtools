13  import static org.biopax.paxtools.io.sif.BinaryInteractionType.COMPONENT_OF;
14  import static org.biopax.paxtools.io.sif.BinaryInteractionType.IN_SAME_COMPONENT;

16  /**
17   * This rule class mines complex membership relations. It mines two types of relations.
18   * COMPONENT_OF: A is a member of the Complex B, membership relation can be multi step.
19   * IN_SAME_COMPONENT: A and B are members of the same complex, and this same complex can be in
20   * different level of nesting.
21   * @author Emek Demir
22   */
23  public class ComponentRule extends InteractionRuleL3Adaptor
24  {
25  	/**
26  	 * List of interaction types mined.
27  	 */
28  	private static List<BinaryInteractionType> binaryInteractionTypes =
29  		Arrays.asList(IN_SAME_COMPONENT);
30  
31  	/**
32  	 * Option to mine IN_SAME_COMPONENT.
33  	 */
34  	private boolean inSameComponent;
35  
36  	/**
37  	 * Option to mine COMPONENT_OF.
38  	 */
39  	private boolean componentOf;
40  
41  	/**
42  	 * Infer interactions where A = the given PhysicalEntity.
43  	 * @param interactionSet to be populated
44  	 * @param pe PhysicalEntity that will be the seed of the inference
45  	 * @param model BioPAX model
46  	 */
47  	public void inferInteractionsFromPE(InteractionSetL3 interactionSet, PhysicalEntity pe,
48  		Model model)
49  	{
50  		if (pe instanceof Complex)
51  		{
52  			Group group = interactionSet.getGroupMap().getMap().get(pe);
53  			if (group != null)
54  			{
55  				Set<EntityReference> members = group.members;
56  				Set<Group> subGroups = group.subgroups;
57  				ArrayList<BioPAXElement> components = new ArrayList<BioPAXElement>(members.size() +
58  					subGroups.size());
59  				components.addAll(members);
60  				components.addAll(subGroups);
61  
62  				if (inSameComponent)
63  				{
64  					BioPAXElement[] sources = group.sources.toArray((BioPAXElement[])
65  						Array.newInstance(BioPAXElement.class, group.sources.size()));
66  
67  					createClique(interactionSet, components,
68  						BinaryInteractionType.IN_SAME_COMPONENT, sources);
69  				}
70  				if (componentOf)
71  				{
72  					for (BioPAXElement component : components)
73  					{
74  
75  						addComponent(component, group, interactionSet);
76  					}
77  				}
78  			}
79  		}
80  	}
81  
82  	/**
83  	 * Adds the component to the group.
84  	 * @param component component to add
85  	 * @param group group to add to
86  	 * @param interactionSet current set of inferred interactions
87  	 */
88  	private void addComponent(BioPAXElement component, Group group, InteractionSetL3 interactionSet)
89  	{
90  		createAndAdd(interactionSet.getGroupMap().getEntityReferenceOrGroup(component),
91  			group,interactionSet, BinaryInteractionType.COMPONENT_OF);
92  	}
93  
94  	/**
95  	 * Initializes options.
96  	 * @param options options map
97  	 */
98  	@Override public void initOptionsNotNull(Map options)
99  	{
100 		inSameComponent = !checkOption(IN_SAME_COMPONENT,Boolean.FALSE,options);
101 		componentOf = !checkOption(COMPONENT_OF,Boolean.FALSE,options);
102 	}
103 
104 	/**
105 	 * Gets supported interaction types.
106 	 * @return interaction types supported
107 	 */
108 	public List<BinaryInteractionType> getRuleTypes()
109 	{
110 		return binaryInteractionTypes;
111 	}
112 }
